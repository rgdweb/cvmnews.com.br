    <script>
        const CONFIG = {
            CHANNEL_ID: 'UCPq2ETz4aAGo2Z-8JisDPIA',
            CHANNEL_HANDLE: '@NobruTV',
            CHANNEL_NAME: 'NobruTV'
        };
        
        class LiveCache {
            constructor() {
                this.cacheTime = 30000;
                this.cacheKey = 'live_' + CONFIG.CHANNEL_ID;
            }
            
            get() {
                try {
                    const cached = localStorage.getItem(this.cacheKey);
                    if (cached) {
                        const data = JSON.parse(cached);
                        if (Date.now() - data.timestamp < this.cacheTime) {
                            return data;
                        }
                    }
                } catch (e) {}
                return null;
            }
            
            set(data) {
                try {
                    localStorage.setItem(this.cacheKey, JSON.stringify({...data, timestamp: Date.now()}));
                } catch (e) {}
            }
        }
        
        class LiveDetector {
            constructor() {
                this.cache = new LiveCache();
                this.isChecking = false;
            }
            
            async detectLive() {
                if (this.isChecking) return null;
                this.isChecking = true;
                
                const cached = this.cache.get();
                if (cached) {
                    this.isChecking = false;
                    return cached;
                }
                
                try {
                    const handleResult = await this.testChannelHandle();
                    if (handleResult) {
                        const result = { status: 'live', videoId: handleResult };
                        this.cache.set(result);
                        this.isChecking = false;
                        return result;
                    }
                    
                    const nameResult = await this.testChannelName();
                    if (nameResult) {
                        const result = { status: 'live', videoId: nameResult };
                        this.cache.set(result);
                        this.isChecking = false;
                        return result;
                    }
                    
                    const embedResult = await this.testMultipleEmbeds();
                    if (embedResult) {
                        const result = { status: 'live', videoId: embedResult };
                        this.cache.set(result);
                        this.isChecking = false;
                        return result;
                    }
                    
                    const result = { status: 'offline', videoId: null };
                    this.cache.set(result);
                    this.isChecking = false;
                    return result;
                    
                } catch (error) {
                    this.isChecking = false;
                    return { status: 'offline', videoId: null };
                }
            }
            
            async testChannelHandle() {
                const urls = [
                    'https://www.youtube.com/' + CONFIG.CHANNEL_HANDLE + '/live',
                    'https://www.youtube.com/c/' + CONFIG.CHANNEL_HANDLE + '/live'
                ];
                
                for (let i = 0; i < urls.length; i++) {
                    try {
                        const result = await this.testEmbed(urls[i]);
                        if (result) return 'live_stream';
                    } catch (e) {}
                }
                return null;
            }
            
            async testChannelName() {
                const urls = [
                    'https://www.youtube.com/@' + CONFIG.CHANNEL_NAME + '/live',
                    'https://www.youtube.com/c/' + CONFIG.CHANNEL_NAME + '/live'
                ];
                
                for (let i = 0; i < urls.length; i++) {
                    try {
                        const result = await this.testEmbed(urls[i]);
                        if (result) return 'live_stream';
                    } catch (e) {}
                }
                return null;
            }
            
            async testMultipleEmbeds() {
                const urls = [
                    'https://www.youtube.com/embed/live_stream?channel=' + CONFIG.CHANNEL_ID,
                    'https://www.youtube.com/channel/' + CONFIG.CHANNEL_ID + '/live'
                ];
                
                for (let i = 0; i < urls.length; i++) {
                    try {
                        const result = await this.testEmbed(urls[i]);
                        if (result) return 'live_stream';
                    } catch (e) {}
                }
                return null;
            }
            
            testEmbed(url) {
                return new Promise(function(resolve) {
                    const iframe = document.createElement('iframe');
                    iframe.style.cssText = 'position:absolute;left:-9999px;width:1px;height:1px;';
                    iframe.src = url + (url.includes('?') ? '&' : '?') + 'autoplay=0&mute=1';
                    
                    let resolved = false;
                    
                    function cleanup() {
                        if (document.body.contains(iframe)) {
                            document.body.removeChild(iframe);
                        }
                    }
                    
                    iframe.onload = function() {
                        if (!resolved) {
                            resolved = true;
                            cleanup();
                            resolve(true);
                        }
                    };
                    
                    iframe.onerror = function() {
                        if (!resolved) {
                            resolved = true;
                            cleanup();
                            resolve(false);
                        }
                    };
                    
                    document.body.appendChild(iframe);
                    
                    setTimeout(function() {
                        if (!resolved) {
                            resolved = true;
                            cleanup();
                            resolve(false);
                        }
                    }, 4000);
                });
            }
        }
        
        class LivePlayer {
            constructor() {
                this.detector = new LiveDetector();
                this.container = document.getElementById('live-player');
                this.currentVideoId = null;
                this.updateInterval = null;
                this.isLiveActive = false;
                
                this.init();
            }
            
            async init() {
                await this.updateLive();
                this.startAutoUpdate();
            }
            
            async updateLive() {
                const result = await this.detector.detectLive();
                
                if (result && result.status === 'live' && result.videoId !== this.currentVideoId) {
                    this.showLivePlayer(result.videoId);
                    this.currentVideoId = result.videoId;
                    this.isLiveActive = true;
                } else if (result && result.status === 'offline' && this.isLiveActive) {
                    this.showWaitingScreen();
                    this.currentVideoId = null;
                    this.isLiveActive = false;
                } else if (!this.isLiveActive && !this.currentVideoId) {
                    this.showWaitingScreen();
                }
            }
            
            showLivePlayer(videoId) {
                const embedUrl = 'https://www.youtube.com/embed/live_stream?channel=' + CONFIG.CHANNEL_ID + '&autoplay=1&mute=0&rel=0&modestbranding=1';
                
                this.container.innerHTML = '<iframe src="' + embedUrl + '" allow="autoplay;encrypted-media;fullscreen"></iframe>';
            }
            
            showWaitingScreen() {
                this.container.innerHTML = '<div class="overlay"><div>Aguardando transmissão...</div><div class="spinner"></div></div>';
            }
            
            startAutoUpdate() {
                const self = this;
                this.updateInterval = setInterval(function() {
                    self.updateLive();
                }, 30000);
            }
        }
        
        let livePlayer;
        
        document.addEventListener('DOMContentLoaded', function() {
            livePlayer = new LivePlayer();
        });
        
        // Evita reiniciar o vídeo na atualização da página
        window.addEventListener('beforeunload', function() {
            if (livePlayer && livePlayer.updateInterval) {
                clearInterval(livePlayer.updateInterval);
            }
        });
        
        // Preserva estado na atualização
        window.addEventListener('pageshow', function(event) {
            if (event.persisted && livePlayer) {
                livePlayer.updateLive();
            }
        });
    </script>
